// src/features/qie/use-agents-registry.ts
import { useEffect, useState } from "react";

export interface RegistryAgent {
  agent_id: string;
  hybrid_pk: { x25519: string; mlkem: string };
  endpoint: string;
  country: string;
  service_type_uri: string;
}

export interface Registry {
  root: string;
  agents: RegistryAgent[];
  leaves: string[];
}

export interface RegistryState {
  status: "loading" | "ready" | "error";
  error?: string;
  root?: string;
  agents?: RegistryAgent[];
  leaves?: string[];
}

export function useAgentsRegistry(url = "/fixtures/qie/qie-agents.json"): RegistryState {
  const [state, setState] = useState<RegistryState>({ status: "loading" });
  useEffect(() => {
    (async () => {
      try {
        const r = await fetch(url);
        if (!r.ok) throw new Error(`HTTP ${r.status}`);
        const j = await r.json();
        if (j.version !== "QIE/1.0") throw new Error(`unsupported version ${j.version}`);
        setState({ status: "ready", root: j.root, agents: j.agents, leaves: j.merkle_tree.leaves });
      } catch (e) {
        setState({ status: "error", error: String(e) });
      }
    })();
  }, [url]);
  return state;
}
```

// src/features/qie/AgentsStep.tsx
import { useState } from "react";
import { useAgentsRegistry, type RegistryAgent } from "./use-agents-registry";
import { useTranslation } from "../../lib/i18n";

export interface AgentsSelection {
  agents: RegistryAgent[];
  threshold: number;
}

export function AgentsStep({ onContinue }: { onContinue: (s: AgentsSelection) => void }) {
  const { t } = useTranslation("escrow");
  const reg = useAgentsRegistry();
  const [picked, setPicked] = useState<Set<string>>(new Set());
  const [threshold, setThreshold] = useState(2);

  if (reg.status === "loading") return <p>{t("setup.loadingRegistry")}</p>;
  if (reg.status === "error") return <p className="text-red-400">{reg.error}</p>;

  const agents = reg.agents ?? [];
  const selected = agents.filter(a => picked.has(a.agent_id));
  const valid = selected.length >= 1 && threshold >= 1 && threshold <= selected.length;

  return (
    <section className="space-y-6">
      <h2 className="font-serif italic text-2xl">{t("setup.agents.title")}</h2>
      <ul className="space-y-2">
        {agents.map(a => (
          <li key={a.agent_id} className="flex gap-3 items-center border border-slate-700 rounded-lg p-3">
            <input type="checkbox" checked={picked.has(a.agent_id)} onChange={e => {
              const next = new Set(picked);
              if (e.target.checked) next.add(a.agent_id); else next.delete(a.agent_id);
              setPicked(next);
            }} />
            <div>
              <div className="font-mono text-sm">{a.agent_id}</div>
              <div className="text-slate-400 text-xs">{a.country} · {a.endpoint}</div>
            </div>
          </li>
        ))}
      </ul>
      <label className="flex items-center gap-3">
        <span>{t("setup.agents.threshold")}</span>
        <input type="number" min={1} max={selected.length || 1} value={threshold}
               onChange={e => setThreshold(Number(e.target.value))}
               className="bg-slate-800 border border-slate-700 rounded px-2 py-1 w-20" />
        <span className="text-slate-400 text-sm">{t("setup.agents.ofN", { n: selected.length })}</span>
      </label>
      <button
        disabled={!valid}
        onClick={() => onContinue({ agents: selected, threshold })}
        className="bg-emerald-500/10 border border-emerald-500/30 px-4 py-2 rounded disabled:opacity-50"
      >{t("setup.next")}</button>
    </section>
  );
}

// src/features/qie/use-escrow-setup.ts
import { useCallback, useState } from "react";
import { buildEscrowConfig, buildEnvelope } from "@qkb/qie-core";
import type { EscrowConfig, HybridPublicKey, ArbitratorRef } from "@qkb/qie-core";
import type { RegistryAgent } from "./use-agents-registry";

function toHex(b: Uint8Array): string { return "0x" + Array.from(b, x => x.toString(16).padStart(2, "0")).join(""); }
function unhex(s: string): Uint8Array { const h = s.replace(/^0x/, ""); const r = new Uint8Array(h.length / 2); for (let i = 0; i < r.length; i++) r[i] = parseInt(h.slice(i * 2, i * 2 + 2), 16); return r; }

export interface SubmitInput {
  pk: `0x04${string}`;
  recovery: Uint8Array;
  agents: RegistryAgent[];
  threshold: number;
  recipientPk: HybridPublicKey;
  arbitrator: ArbitratorRef;
  expiry: number;
}

export function useEscrowSetup() {
  const [state, setState] = useState<{ phase: "idle" | "submitting" | "done" | "error"; escrowId?: string; error?: string }>({ phase: "idle" });

  const submit = useCallback(async (i: SubmitInput) => {
    setState({ phase: "submitting" });
    try {
      const cfg = buildEscrowConfig({
        pk: i.pk,
        agents: i.agents.map(a => ({
          agent_id: a.agent_id,
          hybrid_pk: { x25519: unhex(a.hybrid_pk.x25519), mlkem: unhex(a.hybrid_pk.mlkem) },
          endpoint: a.endpoint,
        })),
        threshold: i.threshold,
        recipient_hybrid_pk: i.recipientPk,
        arbitrator: i.arbitrator,
        expiry: i.expiry,
        jurisdiction: "UA",
      });
      const env = buildEnvelope(cfg, i.recovery);
      for (let idx = 0; idx < i.agents.length; idx++) {
        const a = i.agents[idx];
        const ws = env.wrappedShares[idx];
        const res = await fetch(new URL("/escrow", a.endpoint), {
          method: "POST", headers: { "content-type": "application/json" },
          body: JSON.stringify({
            escrowId: env.escrowId,
            config: env.config,
            ct: {
              kem_ct: { x25519_ct: toHex(ws.ct.kem_ct.x25519_ct), mlkem_ct: toHex(ws.ct.kem_ct.mlkem_ct) },
              wrap: toHex(ws.ct.wrap),
            },
            encR: toHex(env.encR),
          }),
        });
        if (!res.ok) throw new Error(`agent ${a.agent_id} rejected (${res.status})`);
      }
      setState({ phase: "done", escrowId: env.escrowId });
      // NB: registerEscrow on-chain is a separate button shown after acks — avoids implicit wallet prompt.
    } catch (e) {
      setState({ phase: "error", error: String(e) });
    }
  }, []);

  return { state, submit };
}

// src/features/qie/use-escrow-recover.ts
import { useCallback, useState } from "react";
import { reconstructRecovery, hybridDecapsulate, unwrapShare } from "@qkb/qie-core";
import type { EscrowConfig, Evidence, HybridSecretKey } from "@qkb/qie-core";

function toHex(b: Uint8Array): string { return "0x" + Array.from(b, x => x.toString(16).padStart(2, "0")).join(""); }
function unhex(s: string): Uint8Array { const h = s.replace(/^0x/, ""); const r = new Uint8Array(h.length / 2); for (let i = 0; i < r.length; i++) r[i] = parseInt(h.slice(i * 2, i * 2 + 2), 16); return r; }

export function useEscrowRecover() {
  const [state, setState] = useState<{ phase: "idle" | "fetching" | "reconstructing" | "done" | "error"; R?: Uint8Array; error?: string }>({ phase: "idle" });

  const fetchConfig = useCallback(async (escrowId: string, anyAgentUrl: string): Promise<EscrowConfig> => {
    const r = await fetch(new URL(`/escrow/${escrowId}/config`, anyAgentUrl));
    if (!r.ok) throw new Error(`fetch config failed: ${r.status}`);
    return (await r.json()).config;
  }, []);

  const recover = useCallback(async (i: {
    escrowId: string; config: EscrowConfig; evidence: Evidence;
    recipientSk: HybridSecretKey; nonce: string;
  }) => {
    setState({ phase: "fetching" });
    try {
      const shares: Array<{ index: number; value: Uint8Array }> = [];
      let encR: Uint8Array | null = null;
      for (const agent of i.config.agents) {
        const res = await fetch(new URL(`/escrow/${i.escrowId}/release`, agent.endpoint), {
          method: "POST", headers: { "content-type": "application/json" },
          body: JSON.stringify({ evidence: i.evidence, recipient_nonce: i.nonce }),
        });
        if (!res.ok) continue;
        const body = await res.json();
        const wrap = unhex(body.ct.wrap);
        const kem_ct = { x25519_ct: unhex(body.ct.kem_ct.x25519_ct), mlkem_ct: unhex(body.ct.kem_ct.mlkem_ct) };
        const ss = hybridDecapsulate(i.recipientSk, kem_ct);
        const aad = new TextEncoder().encode(i.escrowId + agent.agent_id);
        shares.push(unwrapShare(ss, wrap, aad));
        encR = unhex(body.encR);
        if (shares.length >= i.config.threshold) break;
      }
      if (!encR || shares.length < i.config.threshold) throw new Error("insufficient shares");
      setState({ phase: "reconstructing" });
      const R = reconstructRecovery({ config: i.config, encR }, shares);
      setState({ phase: "done", R });
    } catch (e) {
      setState({ phase: "error", error: String(e) });
    }
  }, []);

  return { state, fetchConfig, recover };
}
```
