**Invariants (PRIVACY.md; binding on all subsequent tasks):**

1. No secret key material — recipient hybrid sk, Phase 1 σ_QES contents, cert_QES contents — is ever written to `localStorage`, `sessionStorage`, `IndexedDB`, or a service worker cache.
2. The Phase 1 binding artifacts and recovery material `R` live in a single React context backed by a `useRef` + plain `useState`. Closing/refreshing the tab discards them. There is no "resume session" button.
3. All sensitive `fetch` calls use `cache: "no-store"` and do not set any cookie.
4. No escrowId, recovery material, or sk ever appears in the URL (no query string, no hash, no history API). Routes with sensitive state keep the state purely in component memory.
5. Any form accepting an sk uses `<input type="password" autoComplete="off" spellCheck={false}>` and is wrapped in a `<form autoComplete="off">`.
6. When sensitive state is in memory, `session-guard.ts` attaches a `beforeunload` handler that prompts the user before tab close/reload (best-effort; browser-controlled).
7. Generated recipient keypairs are offered as a passphrase-encrypted keystore JSON download (scrypt-AES-GCM, parameters frozen in `keystore.ts`). Plain hex view is behind a user-triggered `<details>` reveal with a warning.
8. CSP headers (set at Fly deploy time) disallow `data:` images for any field that could reflect input, disallow inline scripts, and block `connect-src` to anything but the agent endpoints + the configured RPC.

```ts
// src/lib/session-guard.ts
import { useEffect, useRef } from "react";

const tokens = new Set<symbol>();
let installed = false;

function onUnload(e: BeforeUnloadEvent) {
  if (tokens.size === 0) return;
  e.preventDefault();
  e.returnValue = "";
}

function install() {
  if (installed) return;
  installed = true;
  window.addEventListener("beforeunload", onUnload);
}

export function useSensitiveSessionGuard(active: boolean): void {
  const tok = useRef(Symbol("sensitive")).current;
  useEffect(() => {
    install();
    if (active) tokens.add(tok); else tokens.delete(tok);
    return () => { tokens.delete(tok); };
  }, [active, tok]);
}

export function assertNoPersistence(): void {
  // Runtime self-check: log a console error if sensitive keys appear under known storage.
  const offenders = ["qkb.recipientSk", "qie.recipientSk", "qie.bindingSigma"];
  for (const k of offenders) {
    if (localStorage.getItem(k) || sessionStorage.getItem(k)) {
      console.error(`PRIVACY BREAKAGE: ${k} persisted to storage. Clearing.`);
      localStorage.removeItem(k); sessionStorage.removeItem(k);
    }
  }
}
```

**Keystore encoder (scrypt + AES-256-GCM):**

```ts
// src/features/qie/keystore.ts
import { gcm } from "@noble/ciphers/aes";
import { scrypt } from "@noble/hashes/scrypt";
import { randomBytes } from "@noble/hashes/utils";
import type { HybridSecretKey } from "@qkb/qie-core";

const KDF = { N: 1 << 17, r: 8, p: 1, dkLen: 32 } as const;

export interface Keystore {
  version: "QIE-keystore/1";
  kdf: { name: "scrypt"; params: typeof KDF; salt: string };
  cipher: { name: "aes-256-gcm"; iv: string; tag: string };
  ciphertext: string;  // hex of {x25519_sk || mlkem_sk}
}

function toHex(b: Uint8Array): string { return Buffer.from(b).toString("hex"); }
function fromHex(h: string): Uint8Array { return Uint8Array.from(Buffer.from(h, "hex")); }

export async function encryptKeystore(sk: HybridSecretKey, passphrase: string): Promise<Keystore> {
  if (passphrase.length < 12) throw new Error("passphrase too short (min 12 chars)");
  const salt = randomBytes(16);
  const key = scrypt(new TextEncoder().encode(passphrase), salt, KDF);
  const iv = randomBytes(12);
  const plain = new Uint8Array(sk.x25519.length + sk.mlkem.length);
  plain.set(sk.x25519, 0); plain.set(sk.mlkem, sk.x25519.length);
  const ct = gcm(key, iv).encrypt(plain);
  return {
    version: "QIE-keystore/1",
    kdf: { name: "scrypt", params: KDF, salt: toHex(salt) },
    cipher: { name: "aes-256-gcm", iv: toHex(iv), tag: "" },  // tag is last 16 bytes of ct
    ciphertext: toHex(ct),
  };
}

export async function decryptKeystore(ks: Keystore, passphrase: string): Promise<HybridSecretKey> {
  const key = scrypt(new TextEncoder().encode(passphrase), fromHex(ks.kdf.salt), ks.kdf.params);
  const pt = gcm(key, fromHex(ks.cipher.iv)).decrypt(fromHex(ks.ciphertext));
  return { x25519: pt.slice(0, 32), mlkem: pt.slice(32) };
}
```

```ts
// src/features/qie/use-agents-registry.ts
import { useEffect, useState } from "react";

export interface RegistryAgent {
  agent_id: string;
  hybrid_pk: { x25519: string; mlkem: string };
  endpoint: string;
  country: string;
  service_type_uri: string;
}

export interface Registry {
  root: string;
  agents: RegistryAgent[];
  leaves: string[];
}

export interface RegistryState {
  status: "loading" | "ready" | "error";
  error?: string;
  root?: string;
  agents?: RegistryAgent[];
  leaves?: string[];
}

export function useAgentsRegistry(url = "/fixtures/qie/qie-agents.json"): RegistryState {
  const [state, setState] = useState<RegistryState>({ status: "loading" });
  useEffect(() => {
    (async () => {
      try {
        const r = await fetch(url);
        if (!r.ok) throw new Error(`HTTP ${r.status}`);
        const j = await r.json();
        if (j.version !== "QIE/1.0") throw new Error(`unsupported version ${j.version}`);
        setState({ status: "ready", root: j.root, agents: j.agents, leaves: j.merkle_tree.leaves });
      } catch (e) {
        setState({ status: "error", error: String(e) });
      }
    })();
  }, [url]);
  return state;
}
```

```tsx
import { useState } from "react";
import { useAgentsRegistry, type RegistryAgent } from "./use-agents-registry";
import { useTranslation } from "../../lib/i18n";

export interface AgentsSelection {
  agents: RegistryAgent[];
  threshold: number;
}

export function AgentsStep({ onContinue }: { onContinue: (s: AgentsSelection) => void }) {
  const { t } = useTranslation("escrow");
  const reg = useAgentsRegistry();
  const [picked, setPicked] = useState<Set<string>>(new Set());
  const [threshold, setThreshold] = useState(2);

  if (reg.status === "loading") return <p>{t("setup.loadingRegistry")}</p>;
  if (reg.status === "error") return <p className="text-red-400">{reg.error}</p>;

  const agents = reg.agents ?? [];
  const selected = agents.filter(a => picked.has(a.agent_id));
  const valid = selected.length >= 1 && threshold >= 1 && threshold <= selected.length;

  return (
    <section className="space-y-6">
      <h2 className="font-serif italic text-2xl">{t("setup.agents.title")}</h2>
      <ul className="space-y-2">
        {agents.map(a => (
          <li key={a.agent_id} className="flex gap-3 items-center border border-slate-700 rounded-lg p-3">
            <input type="checkbox" checked={picked.has(a.agent_id)} onChange={e => {
              const next = new Set(picked);
              if (e.target.checked) next.add(a.agent_id); else next.delete(a.agent_id);
              setPicked(next);
            }} />
            <div>
              <div className="font-mono text-sm">{a.agent_id}</div>
              <div className="text-slate-400 text-xs">{a.country} · {a.endpoint}</div>
            </div>
          </li>
        ))}
      </ul>
      <label className="flex items-center gap-3">
        <span>{t("setup.agents.threshold")}</span>
        <input type="number" min={1} max={selected.length || 1} value={threshold}
               onChange={e => setThreshold(Number(e.target.value))}
               className="bg-slate-800 border border-slate-700 rounded px-2 py-1 w-20" />
        <span className="text-slate-400 text-sm">{t("setup.agents.ofN", { n: selected.length })}</span>
      </label>
      <button
        disabled={!valid}
        onClick={() => onContinue({ agents: selected, threshold })}
        className="bg-emerald-500/10 border border-emerald-500/30 px-4 py-2 rounded disabled:opacity-50"
```

```ts
export function assembleRecovery(a: Phase1Artifacts): Uint8Array {
  // R = JCS({binding: base64, p7s: base64, cert: base64})
  const obj = {
    binding_b64: Buffer.from(a.bindingJson).toString("base64"),
    cert_b64: Buffer.from(a.certDer).toString("base64"),
    p7s_b64: Buffer.from(a.p7sDer).toString("base64"),
  };
  return new TextEncoder().encode(JSON.stringify(obj));  // JCS import handled by qie-core if needed later
}
```

```ts
// src/features/qie/use-escrow-recover.ts
import { useCallback, useState } from "react";
import { reconstructRecovery, hybridDecapsulate, unwrapShare } from "@qkb/qie-core";
import type { EscrowConfig, Evidence, HybridSecretKey } from "@qkb/qie-core";

function toHex(b: Uint8Array): string { return "0x" + Array.from(b, x => x.toString(16).padStart(2, "0")).join(""); }
function unhex(s: string): Uint8Array { const h = s.replace(/^0x/, ""); const r = new Uint8Array(h.length / 2); for (let i = 0; i < r.length; i++) r[i] = parseInt(h.slice(i * 2, i * 2 + 2), 16); return r; }

export function useEscrowRecover() {
  const [state, setState] = useState<{ phase: "idle" | "fetching" | "reconstructing" | "done" | "error"; R?: Uint8Array; error?: string }>({ phase: "idle" });

  const fetchConfig = useCallback(async (escrowId: string, anyAgentUrl: string): Promise<EscrowConfig> => {
    const r = await fetch(new URL(`/escrow/${escrowId}/config`, anyAgentUrl));
    if (!r.ok) throw new Error(`fetch config failed: ${r.status}`);
    return (await r.json()).config;
  }, []);

  const recover = useCallback(async (i: {
    escrowId: string; config: EscrowConfig; evidence: Evidence;
    recipientSk: HybridSecretKey; nonce: string;
  }) => {
    setState({ phase: "fetching" });
    try {
      const shares: Array<{ index: number; value: Uint8Array }> = [];
      let encR: Uint8Array | null = null;
      for (const agent of i.config.agents) {
        const res = await fetch(new URL(`/escrow/${i.escrowId}/release`, agent.endpoint), {
          method: "POST", headers: { "content-type": "application/json" },
          body: JSON.stringify({ evidence: i.evidence, recipient_nonce: i.nonce }),
        });
        if (!res.ok) continue;
        const body = await res.json();
        const wrap = unhex(body.ct.wrap);
        const kem_ct = { x25519_ct: unhex(body.ct.kem_ct.x25519_ct), mlkem_ct: unhex(body.ct.kem_ct.mlkem_ct) };
        const ss = hybridDecapsulate(i.recipientSk, kem_ct);
        const aad = new TextEncoder().encode(i.escrowId + agent.agent_id);
        shares.push(unwrapShare(ss, wrap, aad));
        encR = unhex(body.encR);
        if (shares.length >= i.config.threshold) break;
      }
      if (!encR || shares.length < i.config.threshold) throw new Error("insufficient shares");
      setState({ phase: "reconstructing" });
      const R = reconstructRecovery({ config: i.config, encR }, shares);
      setState({ phase: "done", R });
    } catch (e) {
      setState({ phase: "error", error: String(e) });
    }
  }, []);

  return { state, fetchConfig, recover };
}
```

```ts
import { useEffect, useState } from "react";
import { createPublicClient, http } from "viem";
import { sepolia } from "viem/chains";
import registryAbi from "../../fixtures/qie/arbitrators/abi/QKBRegistry.abi.json";

export interface EscrowRow {
  pk: `0x04${string}`;
  escrowId: string;
  arbitrator: string;
  expiry: number;
  chainState: "active" | "expired" | "revoked" | "none";
  agentStatuses: { agent_id: string; endpoint: string; status: string }[];
}

export function useEscrowManage(pks: `0x04${string}`[], registryAddr: `0x${string}`): { rows: EscrowRow[]; refresh: () => void } {
  const [rows, setRows] = useState<EscrowRow[]>([]);
  const client = createPublicClient({ chain: sepolia, transport: http() });

  async function load() {
    const out: EscrowRow[] = [];
    for (const pk of pks) {
      const escrowId = await client.readContract({ address: registryAddr, abi: registryAbi, functionName: "escrowCommitment", args: [pk] }) as string;
      if (!escrowId || escrowId === "0x" + "0".repeat(64)) continue;
      const active = await client.readContract({ address: registryAddr, abi: registryAbi, functionName: "isEscrowActive", args: [pk] }) as boolean;
      const entry = await client.readContract({ address: registryAddr, abi: registryAbi, functionName: "escrows", args: [pk] }) as any;
      // Fetch E from any agent to learn endpoints, then poll /status
      // ... (recipient may not have E; use cached session E if Holder, else fetch from known agent endpoints)
      out.push({ pk, escrowId, arbitrator: entry.arbitrator, expiry: Number(entry.expiry), chainState: active ? "active" : entry.revoked ? "revoked" : "expired", agentStatuses: [] });
    }
    setRows(out);
  }
  useEffect(() => { load(); /* eslint-disable-next-line */ }, [pks.join(",")]);
  return { rows, refresh: load };
}
```
