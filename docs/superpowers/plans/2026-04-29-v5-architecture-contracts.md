## §0 — Frozen interface contracts (verbatim from orchestration §2)

### §0.1 — Public-signal layout (14 BN254 field elements)

```
[0]  msgSender              uint256, ≤2^160
[1]  timestamp              uint256, ≤2^64
[2]  nullifier              uint256 (Poseidon₂ output)
[3]  ctxHashHi              uint128
[4]  ctxHashLo              uint128
[5]  bindingHashHi          uint128
[6]  bindingHashLo          uint128
[7]  signedAttrsHashHi      uint128
[8]  signedAttrsHashLo      uint128
[9]  leafTbsHashHi          uint128
[10] leafTbsHashLo          uint128
[11] policyLeafHash         uint256 (already field-domain)
[12] leafSpkiCommit         uint256 (= SpkiCommit(leafSpki))
[13] intSpkiCommit          uint256 (= SpkiCommit(intSpki))
```

The Solidity verifier auto-generated by snarkjs takes `uint[14] input`. Public-signal order MUST match this layout — index drift breaks integration silently.

### §0.2 — `SpkiCommit(spki)` definition

```
1. Parse 91-byte named-curve P-256 DER SPKI → X (32 bytes), Y (32 bytes).
2. Decompose X into 6×43-bit LE limbs:
   X_limbs[i] = (uint256(X) >> (43*i)) & ((1<<43)-1)   for i in [0,5]
3. SpkiCommit = Poseidon₂( Poseidon₆(X_limbs[0..5]),
                           Poseidon₆(Y_limbs[0..5]) )
```

iden3 / circomlib BN254 Poseidon parameters. Your Solidity `P256Verify.spkiCommit(bytes calldata spki)` MUST produce byte-identical output to circuits-eng's TS reference impl on the lead-supplied parity fixture at `fixtures/spki-commit/v5-parity.json`. **The parity Foundry test is the gate.**

### §0.5 — Trust-list and policy-list Merkle leaves

- Trust-list leaves: `SpkiCommit(intSpki)` for each authorized intermediate.
- Policy-list leaves: `policyLeafHash` (the field-domain QKB/2.0 policy commitment).
- Tree depth: 16 for both.
- Hash function at internal nodes: `Poseidon₂(left, right)` (BN254, iden3 params).
- Empty/unused leaves: `Poseidon₁(0)` (matches V4 convention).
- Path encoding: `path[i]` = sibling at level i; `pathBits[i]` = 0 if current node is LEFT child at level i.

// SPDX-License-Identifier: GPL-3.0-or-later
pragma solidity 0.8.24;

/// @notice EIP-7212 P-256 verification + canonical SPKI commitment.
/// @dev    Pure functions; no state. Matches V5 spec §0.2 byte-for-byte.
library P256Verify {
    address internal constant PRECOMPILE = 0x0000000000000000000000000000000000000100;

    /// @dev DER prefix for 91-byte named-curve P-256 SubjectPublicKeyInfo.
    ///      Bytes [0..26]: SEQUENCE + AlgorithmIdentifier(id-ecPublicKey, secp256r1) +
    ///                     BIT STRING header + uncompressed-point prefix.
    bytes27 internal constant SPKI_PREFIX = 0x30593013060_7_2A8648CE3D02010608_2A8648CE3D030107034200_04; // illustrative

    /// @notice Parse a 91-byte named-curve P-256 DER SPKI into raw X, Y coords.
    /// @param  spki The DER bytes (must be exactly 91 bytes with the named-curve prefix).
    /// @return x   32-byte big-endian X coordinate.
    /// @return y   32-byte big-endian Y coordinate.
    function parseSpki(bytes memory spki) internal pure returns (bytes32 x, bytes32 y) {
        require(spki.length == 91, "P256: SPKI length");

        // Verify hardcoded DER prefix bytes [0..26]
        for (uint i = 0; i < 27; i++) {
            require(uint8(spki[i]) == _spkiPrefixByte(i), "P256: SPKI prefix");
        }

        // Extract X from bytes [27..58] and Y from [59..90].
        assembly {
            x := mload(add(spki, 0x3B))  // 0x20 + 27
            y := mload(add(spki, 0x5B))  // 0x20 + 59
        }
    }

    function _spkiPrefixByte(uint i) private pure returns (uint8) {
        // Hardcode the 27-byte prefix for named-curve P-256 SPKI.
        bytes27 prefix = 0x30593013060_7_2A8648CE3D02010608_2A8648CE3D030107034200_04;
        // (Use a literal byte array constant; the inline `bytes27` is illustrative.
        //  Full implementation: store as `bytes` constant or embed via assembly.)
        // Recommend: bytes27 stored inline + bytemask. For clarity:
        bytes memory p = hex"30593013060_7_2A8648CE3D02010608_2A8648CE3D03010703420004";
        return uint8(p[i]);
    }
    // (Worker: clean up the constant representation above; the test asserts
    //  byte-by-byte against the lead-supplied fixture so any implementation
    //  that round-trips correctly is acceptable.)
}

import {PoseidonT3} from "./PoseidonT3.sol";
import {PoseidonT7} from "./PoseidonT7.sol";

function spkiCommit(bytes memory spki) internal pure returns (uint256) {
    (bytes32 x, bytes32 y) = parseSpki(spki);
    uint256[6] memory xLimbs = _decomposeTo643Limbs(x);
    uint256[6] memory yLimbs = _decomposeTo643Limbs(y);
    uint256 hashX = PoseidonT7.poseidon(xLimbs);
    uint256 hashY = PoseidonT7.poseidon(yLimbs);
    return PoseidonT3.poseidon([hashX, hashY]);
}

uint256 private constant LIMB_MASK = (1 << 43) - 1;

function _decomposeTo643Limbs(bytes32 v) private pure returns (uint256[6] memory limbs) {
    uint256 vU = uint256(v);
    for (uint i = 0; i < 6; i++) {
        limbs[i] = (vU >> (43 * i)) & LIMB_MASK;
    }
}

// src/libs/PoseidonMerkle.sol
import {PoseidonT3} from "./PoseidonT3.sol";

library PoseidonMerkle {
    /// @notice Verify a 16-deep Merkle membership proof under Poseidon₂ hashing.
    /// @param leaf      The claimed leaf value.
    /// @param path      Sibling hashes at each of the 16 levels.
    /// @param pathBits  Bit i = 0 means current node is LEFT child at level i.
    /// @param root      The expected root.
    function verify(
        bytes32 leaf,
        bytes32[16] memory path,
        uint256 pathBits,
        bytes32 root
    ) internal pure returns (bool) {
        uint256 cur = uint256(leaf);
        for (uint i = 0; i < 16; i++) {
            uint256 sibling = uint256(path[i]);
            if (((pathBits >> i) & 1) == 0) {
                cur = PoseidonT3.poseidon([cur, sibling]);
            } else {
                cur = PoseidonT3.poseidon([sibling, cur]);
            }
        }
        return cur == uint256(root);
    }
}

import {IQKBRegistry} from "./IQKBRegistry.sol";

interface IGroth16VerifierV5 {

        uint[2] calldata a,
        uint[2][2] calldata b,
        uint[2] calldata c,

contract QKBRegistryV5 is IQKBRegistry {
    IGroth16VerifierV5 public immutable groth16Verifier;
    address public admin;
    bytes32 public override trustedListRoot;
    bytes32 public policyRoot;
    uint256 public constant MAX_BINDING_AGE = 1 hours;

    mapping(address => bytes32) public override nullifierOf;
    mapping(bytes32 => address) public registrantOf;

    event Registered(address indexed holder, bytes32 indexed nullifier, uint256 timestamp);
    event TrustedListRootRotated(bytes32 indexed previous, bytes32 indexed current, address admin);
    event PolicyRootRotated(bytes32 indexed previous, bytes32 indexed current, address admin);

    modifier onlyAdmin() {
        require(msg.sender == admin, "QKBV5: only admin");
        _;
    }

    constructor(

        address _admin,

        trustedListRoot = _initialTrustedListRoot;

        emit TrustedListRootRotated(trustedListRoot, newRoot, msg.sender);
        trustedListRoot = newRoot;

        emit PolicyRootRotated(policyRoot, newRoot, msg.sender);
        policyRoot = newRoot;
